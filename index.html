4 Пространства имен (namespaces) 

В данном видеоуроки автор рассказывает про различие 
локальной переменной и глобальной также что можно 
создать свою пространство имен через namespaces 
и выводить это

Область видимости - можем воспринимать как некоторое 
пространство, ограниченное {} (далее фигурные скобки) 
и в котором можем объявить переменные и т.д

Функция main() - главная точка входа в программу, 
с которой начинает выполнение программы и у неё 
своя область видимости, ограниченная с {}

Глобальная область видимости - это вся остальная 
область видимости вне любой функции

namespace - пространство имен, создаем свою 
область видимости

using namespace - чтобы получить к своему 
созданному пространству имён и проще можем 
обращаться к своим переменным

						
						
#include 

namespace outNameSpace {

	int var1 = 25;
	double var2 = 6.67;
	bool flag = true;
	char varCh = 'y';

	namespace outNamespaceInner {
		int valueInner = 1465;
	}

}

int value = 500; // Глобальная переменная

//std::cout << value std::endl;

int main()
{
	using namespace outNameSpace;
	namespace ns = outNameSpace::outNamespaceInner; // упращенный доступ к своим namespace
	std::cout << flag << std::endl;
	std::cout << var2 << std::endl;
	std::cout << varCh << std::endl;

	std::cout << outNameSpace::var1 << std::endl; // Доступ к своему пространству имён 
	//std::cout << outNameSpace::outNamespaceInner::valueInner << std::endl; // Доступ к вложенному namespace
	std::cout << ns::valueInner << std::endl; // // упращенный доступ к своим namespace

	int value = 100;
	//std::cout << value << std::endl;
	{
		int value = 300;
		// Если убрать int, то повлияем на перменную сверху и выйдет, 300
		std::cout << value << std::endl;
	}

	std::cout << value << std::endl;

	std::cout << ::value << std::endl; // Обращание к глобальная переменной и вывод

	return 0;
}
